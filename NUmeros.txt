un programa que al ingresar un numero de 2 digitos sume sus digitos 
#include <iostream>
using namespace std;
int main (){
int num,x,y;
cin>>num;
if(num<0){
num=num*-1;}

if(num>=10 and num<=99){
    x=num/10;
y=num%10;
cout<<x+y;}
else
    cout<<"ingrese un numero de dos digitos";


return 0;
}
